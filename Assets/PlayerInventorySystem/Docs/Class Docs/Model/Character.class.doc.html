Character Struct

Struct that outlines the base character stats.

Public Fields

characterName

The name of the character.

public string characterName;

ID

The unique identifier for the character.

public int ID;

GENDER

The gender of the character.

public GENDER GENDER;

Level

The level of the character.

public int Level;

Experience

The experience points of the character.

public int Experience;

The health points of the character.

public float Health;

Mana

The mana points of the character.

public float Mana;

Stamina

The stamina points of the character.

public float Stamina;

Strength

The strength attribute of the character.

public float Strength;

Dexterity

The dexterity attribute of the character.

public float Dexterity;

IQ

The intelligence (IQ) attribute of the character.

public float IQ;

Armor

The armor attribute of the character.

public float Armor;

Speed

The speed attribute of the character.

public float Speed;

Luck

The luck attribute of the character.

public float Luck;

Public Properties

Damage

The calculated damage value based on strength, speed, dexterity, intelligence, and luck.

public float Damage
{
    get
    {
        return ((Strength * IQ) + (Dexterity * Speed)) * Luck;
    }
}

Usage Examples

Example: Create a new character

Character newCharacter = new Character
{
    characterName = "Archer",
    ID = 1,
    GENDER = GENDER.Male,
    Level = 5,
    Experience = 1500,
    Health = 100f,
    Mana = 50f,
    Stamina = 75f,
    Strength = 20f,
    Dexterity = 30f,
    IQ = 25f,
    Armor = 10f,
    Speed = 15f,
    Luck = 5f
};

Debug.Log("Character Name: " + newCharacter.characterName);
Debug.Log("Character Damage: " + newCharacter.Damage);
