ItemData Class

Defines an item's stats in the Item catalog. Be careful when changing these values as it will affect all items in the catalog and may cause data loss.

Public Fields

name

The human-readable name of this object.

public string name;

id

The catalog ID of this item.

public int id;

description

A brief description of this item.

public string description = "";

sprite

The inventory sprite used to display this item in slots.

public Sprite sprite;

BaseMaterial

The base material type of the item.

public MATERIALTYPE BaseMaterial;

MaterialVariant

The material variant of the item.

public Enum MaterialVariant;

prefabSingle

The prefab that represents a single instance of the item in the game world.

public GameObject prefabSingle;

prefabMultiple

The prefab that represents a stack of the item in the game world.

public GameObject prefabMultiple;

worldPrefab

The game object that represents this item when equipped on the player or placed in the game world.

public GameObject worldPrefab;

itemType

Defines the item's type (e.g., Consumable, Usable, Wearable, Quest).

public ITEMTYPE itemType;

slotType

Defines the type of slot this item requires.

public SLOTTYPE slotType;

ConsumeOnPickup

Indicates if this item will be consumed/used when picked up. The item will not be placed in the inventory.

public bool ConsumeOnPickup = false;

salvageable

Indicates if this item can be salvaged from other items.

public bool salvageable = true;

recycable

Indicates if this item can be recycled into other items via the salvage process.

public bool recycable = true;

recipe

The recipe to craft this item.

public Recipe recipe = new Recipe();

craftCount

The number of this item that the crafting recipe will yield.

public int craftCount = 1;

maxStackSize

The max number of this item that can be stacked in one slot.

public int maxStackSize = 5;

maxDurability

The durability of this item when new.

public float maxDurability = 0;

damage

The damage bonus this item will bestow upon the player when equipped.

public float damage { get; }

Strength

The strength bonus this item will bestow upon the player when equipped.

public float Strength;

Speed

The speed bonus this item will bestow upon the player when equipped.

public float Speed;

health

The health bonus this item will bestow upon the player when equipped.

public float health;

stamina

The stamina bonus this item will bestow upon the player when equipped.

public float stamina;

mana

The mana bonus this item will bestow upon the player when equipped.

public float mana;

armor

The armor bonus this item will bestow upon the player when equipped.

public float armor;

IQ

The IQ/Intelligence bonus this item will bestow upon the player when equipped.

public float IQ;

Dexterity

The dexterity bonus this item will bestow upon the player when equipped.

public float Dexterity;

Luck

The luck bonus this item will bestow upon the player when equipped.

[Range(0f, 1f)]
public float Luck;

Usage Examples

Example: Create a new item and set its properties

ItemData newItemData = new ItemData();
newItemData.name = "Sword of Destiny";
newItemData.id = 1;
newItemData.description = "A powerful sword with magical properties.";
newItemData.sprite = mySprite;
newItemData.BaseMaterial = MATERIALTYPE.Metal;
newItemData.prefabSingle = myPrefabSingle;
newItemData.prefabMultiple = myPrefabMultiple;
newItemData.worldPrefab = myWorldPrefab;
newItemData.itemType = ITEMTYPE.Usable;
newItemData.slotType = SLOTTYPE.Weapon;
newItemData.ConsumeOnPickup = false;
newItemData.salvageable = true;
newItemData.recycable = true;
newItemData.recipe = myRecipe;
newItemData.craftCount = 1;
newItemData.maxStackSize = 10;
newItemData.maxDurability = 100f;
newItemData.Strength = 10f;
newItemData.Speed = 5f;
newItemData.health = 20f;
newItemData.stamina = 15f;
newItemData.mana = 0f;
newItemData.armor = 5f;
newItemData.IQ = 3f;
newItemData.Dexterity = 7f;
newItemData.Luck = 0.5f;

Example: Calculate the damage of an item

float itemDamage = newItemData.damage;
Debug.Log("Item Damage: " + itemDamage);
